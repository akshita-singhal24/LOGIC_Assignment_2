  requires n >= 0
  decreases n
{

}

method Factorial(n: int) returns (res: int)
  requires n >= 0
  ensures res == Fact(n)
{

  var i := 0;
  while i < n

    invariant res == Fact(i)
    decreases n - i
  {
    i := i + 1;
    res := res * i;
  }
}

Task 4 – Tribonacci Sequence

Test Case: Tribonacci Sequence 
function Trib(n: int): int
  requires n >= 0
  decreases n
{
  if n == 0 then 0
  else if n == 1 then 1
  else if n == 2 then 1
  else Trib(n-1) + Trib(n-2) + Trib(n-3)
}

method Tribonacci(n: int) returns (result: int)
  requires n >= 0
  ensures result == Trib(n)
{
  if n == 0 { return 0; }
  if n == 1 { return 1; }
  if n == 2 { return 1; }

  var a := 0; // T(0)
  var b := 1; // T(1)
  var c := 1; // T(2)
  var i := 2;
  while i < n
    invariant 2 <= i <= n
    invariant a == Trib(i-2)
    invariant b == Trib(i-1)
    invariant c == Trib(i)
    decreases n - i
  {
    i := i + 1;
    var next := a + b + c;
    a := b;
    b := c;
    c := next;
  }
  return c;
}
Loop Invariants:
  • a == Trib(i-2), b == Trib(i-1), c == Trib(i): Recent values tracked for proof.
